#!/usr/bin/python
# Filename: using_file.py

poem = '''\
Programming is fun
When the work is done
if you wanna make your work also fun:
        use Python!
'''

f = file('poem.txt', 'w') # 读模式 r，写模式w，追加模式a
f.write(poem) # write text to file
f.close() # close the file

f = file('poem.txt')

    line = f.readline()

f.close() # close the file
